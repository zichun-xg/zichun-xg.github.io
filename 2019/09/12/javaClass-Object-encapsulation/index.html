    封装

    private:其修饰的成员为私有的只能在本类中相互调用
    对私有的属性 提供对外的公共的访问方法
    get()
    set()
    称为setter以及getter方法

    代码示例 Animal封装

    package com.javasm.odj;

    public class Animal {
    private String name;
    private String breed;
    private String cloor;
    public String getName() {
    return name;
    }
    public void setName(String name) {
    this.name = name;
    }
    public String getBreed() {
    return breed;
    }
    public void setBreed(String breed) {
    this.breed = breed;
    }
    public String getCloor() {
    return cloor;
    }
    public void setCloor(String cloor) {
    this.cloor = cloor;
    }
    public void run() {
    System.out.println(name + “在跑”);
    }
    public void show() {
    System.out.println(“这是一只” + cloor + “的” + breed + “,它叫” + name);
    }
    public Animal() {//无参构造
    }
    public Animal(String name, String breed, String cloor) {
    this.name = name;
    this.breed = breed;
    this.cloor = cloor;
    }
    }

    代码示例 创建对象

    package com.javasm.odj;

    public class AnimalText {
    public static void main(String[] args) {
    Animal dog = new Animal();//创建一个dog对象
    dog.setName(“辣椒”);
    dog.setBreed(“二哈”);
    dog.setCloor(“灰色”);
    dog.show();
    dog.run();
    Animal cat = new Animal();//创建一个cat对象
    cat.setName(“香蕉”);
    cat.setBreed(“加菲猫”);
    cat.setCloor(“黄色”);
    cat.show();
    Animal birds = new Animal(“花花”,”鹦鹉”,”白色”);
    birds.show();
    System.out.println(birds.getName());
    }
    }

    运行结果
